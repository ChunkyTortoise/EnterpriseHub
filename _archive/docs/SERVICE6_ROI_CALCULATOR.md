#### Calculation Engine
```javascript
// Core ROI calculation function
function calculateROI(inputs) {
  const timeSavings = calculateTimeSavings(inputs);
  const revenueIncrease = calculateRevenueIncrease(inputs);
  const costReduction = calculateCostReduction(inputs);
  const totalBenefits = timeSavings + revenueIncrease + costReduction;
  
  const initialCost = 2400;
  const monthlyCosts = calculateMonthlyCosts(inputs);
  const annualCosts = initialCost + (monthlyCosts * 12);
  
  const roi = ((totalBenefits - annualCosts) / annualCosts) * 100;
  const paybackPeriod = initialCost / (totalBenefits / 12 - monthlyCosts);
  
  return {
    roi: roi,
    paybackDays: Math.ceil(paybackPeriod * 30),
    annualBenefits: totalBenefits,
    annualCosts: annualCosts
  };
}
```
